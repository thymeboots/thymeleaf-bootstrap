        <tb:button target="#" look="primary" value="Save" ></tb:button>

    </tb:facet>
</tb:modal>
Attribute Default value Description
backdrop true false:static modal
centered false true:modal will be centered
closable true false:do not show close button on header
footer footer text
header header text
id modal id
look secondary modal header look type. can be success, secondary,warning, danger, primary, dark, info etc
scrollablefalse true:modal content scrollable
